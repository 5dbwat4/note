<PTAQuiz programming title="A1 Abstract Factory Maze Game">
  <Q>
    We are designing a Maze Game using the **Abstract Factory** pattern. This design pattern allows us to switch between different maze themes, such as a Standard Maze, an Enchanted Maze, or a Dungeon Maze, without changing the core maze-building logic.

    Read the following framework code:

    ```cpp
    #include <iostream>
    #include <vector>
    using namespace std;

    enum Direction { North, South, East, West };

    class MapSite {
    public:
        virtual void Enter() = 0;
        virtual ~MapSite() {}
    };

    class Room : public MapSite {
    protected:
        int _roomNumber;
        MapSite* _sides[4];
    public:
        Room(int n) : _roomNumber(n) {
            for (int i = 0; i < 4; ++i)
                _sides[i] = NULL;
        }
        void SetSide(Direction d, MapSite* s) { _sides[d] = s; }
        MapSite* GetSide(Direction d) const { return _sides[d]; }
        virtual void Enter() {
            cout << "Enter Standard Room " << _roomNumber << endl;
        }
    };

    class Door : public MapSite {
    protected:
        Room* _room1;
        Room* _room2;
    public:
        Door(Room* r1, Room* r2) : _room1(r1), _room2(r2) {}
        virtual void Enter() {
            cout << "Open a Standard Door" << endl;
        }
    };

    class Maze {
        vector<Room*> _rooms;
        vector<Door*> _doors;
    public:
        void AddRoom(Room* r) { _rooms.push_back(r); }
        void AddDoor(Door* d) { _doors.push_back(d); }

        size_t GetRoomCount() const { return _rooms.size(); }
        Room* GetRoom(size_t index) const {
            if (index < _rooms.size()) return _rooms[index];
            return NULL;
        }

        ~Maze() {
            for (size_t i = 0; i < _rooms.size(); ++i) {
                delete _rooms[i];
            }
            for (size_t i = 0; i < _doors.size(); ++i) {
                delete _doors[i];
            }
        }
    };

    class MazeFactory {
    public:
        virtual Maze* MakeMaze() const { return new Maze(); }
        virtual Room* MakeRoom(int n) const { return new Room(n); }
        virtual Door* MakeDoor(Room* r1, Room* r2) const { return new Door(r1, r2); }
        virtual ~MazeFactory() {}
    };
    ```

    Requirements:
