        char grade;
    
    public:
        GroupB(const string& n, const string& nm, int a, int b, char g)
            : Student(n, nm), s1(a), s2(b), grade(g) {
            double avg = (s1 + s2) / 2.0;
            if (avg > awardLine) awardLine = avg;

            for (int i = 0; i < 5; ++i) sum += score[i];
