  <Q>
    How many times is the destructor of class `A` called during execution?

    ```cpp
    #include <iostream>
    using namespace std;

    class A {
     public:
      A() { cout << "A()" << endl; }
      ~A() { cout << "~A()" << endl; }
    };

    void foo(A a) {
      A arr[5];
      A *p = new A[3];
      throw p;
    }

    int main()
    {
      try {
        A a;
        foo(a);
      } catch (A *p) {
        delete[] p;
        cout << "catched" << endl;
      }
    }
    ```
  </Q>

  <C>`5`</C>
  <C>`7`</C>
  <C>`10`</C>
  <C>`3`</C>

  <Explanation>
    `main` 中的局部对象 `a` 析构 `1` 次；`foo(a)` 的按值参数 `a` 析构 `1` 次；`foo` 中局部数组 `arr[5]` 在异常传播时析构 `5` 次；捕获到动态数组指针后 `delete[] p` 析构 `3` 次。合计 `1 + 1 + 5 + 3 = 10`。
  </Explanation>
</PTAQuiz>

    void parseattribute(ifstream& ifs, std::vector<shape*> &vShape)
    {
        shape* pX = new X();
        if (!pX->parseattribute(ifs))
        {

            return;

        vShape.push_back(pX);

        std::string str;

        while (1)

            str = "";
            ifs >> str;
            if (str == "Rectangle")
                parseattribute<Rectangle>(ifs, m_vShape);
            else if (str == "Circle")
                parseattribute<Circle>(ifs, m_vShape);
            else if (str == "Ellipse")

  <A>
    One possible implementation is:

    ```cpp
    class Rectangle : public shape
    {
    public:
        bool parseattribute(ifstream& ifs) override
        {
            return static_cast<bool>(ifs >> positionLabel >> x >> y >> sizeLabel >> width >> height);
        }

        void draw() override
        {
            cout << "Rectangle " << x << " " << y << " "
                 << sizeLabel << " " << width << " " << height << endl;
        }

    private:
        string positionLabel;
        string sizeLabel;
        double x = 0;
        double y = 0;
        double width = 0;
        double height = 0;
    };

    class Circle : public shape
    {
    public:
        bool parseattribute(ifstream& ifs) override
        {
            return static_cast<bool>(ifs >> centerLabel >> x >> y >> radiusLabel >> radius);
        }

        void draw() override
        {
            cout << "Circle " << x << " " << y << " "
                 << radiusLabel << " " << radius << endl;
        }

    private:
        string centerLabel;
        string radiusLabel;
        double x = 0;
        double y = 0;
        double radius = 0;
    };

    class Ellipse : public shape
    {
    public:
        bool parseattribute(ifstream& ifs) override
        {
            return static_cast<bool>(ifs >> centerLabel >> x >> y >> radiusLabel >> radiusX >> radiusY);
        }

        void draw() override
        {
            cout << "Ellipse " << x << " " << y << " "
                 << radiusLabel << " " << radiusX << " " << radiusY << endl;
        }

    private:
        string centerLabel;
        string radiusLabel;
        double x = 0;
        double y = 0;
        double radiusX = 0;
        double radiusY = 0;
    };
    ```

    Each concrete shape class implements the same two virtual functions. `parseattribute()` consumes exactly the tokens that follow the shape type in the input file, stores the numeric attributes, and reports whether parsing succeeded. `draw()` prints the stored data in the format required by the sample output. Since `CShapeDocument` stores `shape*`, the virtual functions provide the polymorphic behavior needed for parsing and drawing different shapes through the base interface.
