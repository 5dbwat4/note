    };

        } catch (FileError& fe) {

            throw;
        }
    }

    int main() {
        try {
            ProcessFile("goodfile.txt");
            ProcessFile("badfile.txt");

            cout << "Main exception caught" << endl;
        }
        return 0;

  <Explanation>

  <Q>
    What are the outputs of the following code?

    ```cpp
    #include <iostream>
    #include <cstring>
    using namespace std;

    class DynamicArray {
    public:
        DynamicArray(int size = 0) : size(size), data(new int[size]) {
            for (int i = 0; i < size; ++i) {
                data[i] = i + 1;
            }
        }

        DynamicArray(const DynamicArray& other) : size(other.size), data(new int[other.size]) {
            for (int i = 0; i < other.size; ++i) {
                data[i] = other.data[i];
            }
        }

        DynamicArray& operator=(const DynamicArray& other) {
            if (this != &other) {
                delete[] data;
                size = other.size;
                data = new int[size];
                for (int i = 0; i < size; ++i) {
                    data[i] = other.data[i];
                }
            }
            return *this;
        }

        DynamicArray operator+(const DynamicArray& other) {
            DynamicArray result(size + other.size);
            for (int i = 0; i < size; ++i) {
                result.data[i] = data[i];
            }
            for (int i = 0; i < other.size; ++i) {
                result.data[size + i] = other.data[i];
            }
            return result;
        }

        int& operator[](int index) {
            return data[index];
        }

        friend ostream& operator<<(ostream& os, const DynamicArray& arr) {
            for (int i = 0; i < arr.size; ++i) {
                os << arr.data[i] << (i < arr.size - 1 ? ", " : "");
            }
            return os;
        }

        ~DynamicArray() {
            delete[] data;
        }

    private:
        int size;
        int* data;
    };

    int main() {
        DynamicArray arr1(3);
        DynamicArray arr2(2);
        DynamicArray arr3;

        arr3 = arr1 + arr2;

        arr3[2] = 10;
        cout << arr3[0] << endl;
        cout << arr3[1] << endl;
        cout << arr3[2] << endl;
        cout << arr3[4] << endl;

        return 0;
    }
    ```
  </Q>

  输出第 1 行：<Blank>{`1`}</Blank>

  输出第 2 行：<Blank>{`2`}</Blank>

  输出第 3 行：<Blank>{`10`}</Blank>

  输出第 4 行：<Blank>{`2`}</Blank>

  <Explanation>
    `arr1` 初始为 `[1, 2, 3]`，`arr2` 初始为 `[1, 2]`。`arr1 + arr2` 连接成 `[1, 2, 3, 1, 2]` 并赋给 `arr3`。随后 `arr3[2] = 10`，所以 `arr3` 变为 `[1, 2, 10, 1, 2]`。
  </Explanation>

<PTAQuiz programming title="主观题：Online Support Chat System">
  <Q>
    We aim to design an **Online Support Chat System** that routes customer queries to appropriate support agents based on their expertise. The system follows the **Chain of Responsibility Design Pattern**, allowing queries to be processed step by step through a chain of handlers.

    There are `6` classes in total in the system:

    - `SupportQuery`: a class representing a customer query, with attribute `type`, such as `"Billing"`, `"Technical"`, or `"Customer Service"`.
    - `Handler`: an abstract class representing a generic handler in the chain. It declares `SetNext(Handler* handler)` and `Handle(SupportQuery& query)`.
    - `BaseHandler`: a class derived from `Handler` implementing default chaining behavior.
    - `BillingHandler`: a concrete handler that processes `"Billing"` queries.
    - `TechnicalHandler`: a concrete handler that processes `"Technical"` queries.
    - `CustomerServiceHandler`: a concrete handler that processes `"Customer Service"` queries.

    The implementation for `BaseHandler` and `BillingHandler` is as follows:

    ```cpp
    // Implements chaining logic class
    // BaseHandler implements chaining logic
    class BaseHandler : public Handler {
    private:
        Handler* nextHandler = nullptr;
    public:
        Handler* SetNext(Handler* handler) override {
            nextHandler = handler;
            return handler;
        }
        string Handle(SupportQuery& query) override {
            if (nextHandler) {
                return nextHandler->Handle(query);
            }
            return {};
        }
    };

    // Concrete Handlers
    class BillingHandler : public BaseHandler {
    public:
        string Handle(SupportQuery& query) override {
            if (query.getType() == "Billing") {
                return "BillingHandler: Handling Billing query.\n";
            } else {
                return BaseHandler::Handle(query);
            }
        }
    };
    ```

    Your task is to implement `SupportQuery`, `Handler`, `TechnicalHandler`, and `CustomerServiceHandler`, in order to make sure that the following code works properly.

    ```cpp
    #include <iostream>
    #include <string>
    #include <vector>
    using namespace std;

    // Client code to test the chain
    void ClientCode(Handler& handler, vector<SupportQuery>& queries) {
        for (auto& query : queries) {
            cout << "Processing query (Type: " << query.getType() << ")\n";
            const string result = handler.Handle(query);
            if (!result.empty()) {
                cout << "  " << result;
            } else {
                cout << "  " << query.getType() << " query was left unhandled.\n";
            }
        }
    }

    int main() {
        // Create handlers
        BillingHandler* billing = new BillingHandler;
        TechnicalHandler* technical = new TechnicalHandler;
        CustomerServiceHandler* customerService = new CustomerServiceHandler;

        // Build the chain
        billing->SetNext(technical)->SetNext(customerService);

        // Test queries
        vector<SupportQuery> queries = {
            SupportQuery("Billing"),
            SupportQuery("Technical"),
            SupportQuery("Customer Service")
        };
        cout << "Chain: Billing > Technical > Customer Service" << endl;
        ClientCode(*billing, queries);
        cout << endl;
        cout << "Subchain: Technical > Customer Service" << endl;
        ClientCode(*technical, queries);

        // Clean up
        delete billing;
        delete technical;
        delete customerService;
        return 0;
    }
    ```

    The output of the program will be:

    ```txt
    Chain: Billing > Technical > Customer Service
    Processing query (Type: Billing)
      BillingHandler: Handling Billing query.
    Processing query (Type: Technical)
      TechnicalHandler: Handling Technical query.
    Processing query (Type: Customer Service)
      CustomerServiceHandler: Handling Customer Service query.

    Subchain: Technical > Customer Service
    Processing query (Type: Billing)
      Billing query was left unhandled.
    Processing query (Type: Technical)
      TechnicalHandler: Handling Technical query.
    Processing query (Type: Customer Service)
      CustomerServiceHandler: Handling Customer Service query.
    ```

  <A>
    One possible implementation is:

    ```cpp
    class SupportQuery {
    private:
        string type;

    public:
        explicit SupportQuery(const string& type) : type(type) {}

        string getType() const {
            return type;
        }
    };

    class Handler {
    public:
        virtual Handler* SetNext(Handler* handler) = 0;
        virtual string Handle(SupportQuery& query) = 0;
        virtual ~Handler() = default;
    };

    class TechnicalHandler : public BaseHandler {
    public:
        string Handle(SupportQuery& query) override {
            if (query.getType() == "Technical") {
                return "TechnicalHandler: Handling Technical query.\n";
            }
            return BaseHandler::Handle(query);
        }
    };

    class CustomerServiceHandler : public BaseHandler {
    public:
        string Handle(SupportQuery& query) override {
            if (query.getType() == "Customer Service") {
                return "CustomerServiceHandler: Handling Customer Service query.\n";
            }
            return BaseHandler::Handle(query);
        }
    };
    ```

    `SupportQuery` only needs to store and expose the query type. `Handler` supplies the polymorphic interface required by `BaseHandler` and all concrete handlers. Each concrete handler checks whether it can process the current type; if it cannot, it delegates to `BaseHandler::Handle(query)`, which forwards the request to the next handler or returns an empty string when the chain ends.
  </A>
